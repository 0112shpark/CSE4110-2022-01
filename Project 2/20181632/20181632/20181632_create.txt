create table Customer (Customer_ID numeric(5,0), C_name varchar(7),C_address varchar(20), primary key (Customer_ID))
create table Company (Company_Name varchar(20), account numeric(10,0), primary key (Company_Name))
create table Product (Product_name varchar(15), Company_Name varchar(10), price numeric(5,0), primary key (Product_name),foreign key (Company_Name) references Company (Company_Name) on update cascade)
create table Store(S_location varchar(20), S_dollar_sold numeric(10,0), primary key (S_location))
create table Store_inventory(Product_name  varchar(15), S_location varchar(20), S_inventory  numeric(4,0), primary key (Product_name,S_location), foreign key (S_location) references Store (S_location)on delete cascade)
create table Bill (Customer_ID numeric(5,0),Company_Name varchar(20), primary key (Customer_ID,Company_Name))
create table Customer_phone(Customer_ID numeric(5,0), Phone_number numeric(8,0), primary key (Customer_ID,Phone_number), foreign key (Customer_ID) references Customer (Customer_ID) on delete cascade )
create table Warehouse(Product_name varchar(15), W_location varchar(20), W_inventory numeric(4,0), Company_Name varchar(10), PRIMARY KEY (Product_name, W_location),  foreign key (Company_Name) references Company (Company_Name)on update cascade)
create table Ship(Tracking_number numeric(5,0), S_company varchar(10) not null, stat varchar(10), Customer_ID numeric(5,0), Product_name varchar(15), W_location varchar(20), primary key (Tracking_number), foreign key (Customer_ID) references Customer (Customer_ID) on delete cascade, foreign key (Product_name,W_location) references Warehouse (Product_name,W_location) on update cascade)
create table Order_info (On_Off varchar(10), Product_name varchar(15), Customer_ID numeric(5,0), O_year numeric(4,0) check (O_year > 2020 and O_year < 2023), O_month numeric(2,0) check (O_month > 0 and O_month < 13), O_date numeric(2,0) check (O_date > 0 and O_date < 32), Tracking_number numeric(5,0), S_location  varchar(20), amount numeric(2,0), primary key (On_Off,Product_name,Customer_ID,O_date,O_month,O_year), foreign key (Product_name) references Product (Product_name) on delete cascade, foreign key (Customer_ID) references Customer (Customer_ID) on delete cascade, foreign key (Tracking_number) references Ship (Tracking_number) on update cascade, foreign key (S_location) references Store (S_location) on update cascade);
